# Radien des groen und kleinen Halbkreises
r = 0.1
R = 10

def create_nyquist_path_segments(P, r, R, n=100) -> list:
    """
    Diese Funktion erstellt eine Liste von Arrays. Jedes Array enthlt
    `n` komplexe Zahlen und reprsentiert einen Abschnitt (Gerade oder
    Halbkreis) des gesamten Nyquistpfades.
    Die Funktion erlaubt P mit einer beliebigen Anzahl rein imaginrer
    Polstellen.
    
    :param P:    bertragungsfunktion als tf-Objekt
    :param r:    Radius der kleinen Halbkreise um Pole
    :param R:    Radius des groen schlieenden Halbkreises
    :param n:    Anzahl der Punkte in jedem Abschnitt
    """
    poles = P.pole()
    # die Pole auswhlen, deren Realteil = 0 ist
    imag_poles = poles[np.real(poles)==0]
    
    # Halbkreise um die imaginren Pole erstellen
    semi_circles = []
    for pole in imag_poles:
        semi_circle = pole + r*exp(1j*pi*np.linspace(1.5, 0.5, n))
        semi_circles.append(semi_circle)
        
    # finalen groen Halbkreis erstellen
    semi_circle = R*exp(1j*pi*np.linspace(0.5, -0.5, n))
    semi_circles.append(semi_circle)
        
    # Abschnitte auf der imaginren Achse
    line_segment_points = [-1j*R, 1j*R]
    for pole in imag_poles:
        line_segment_points.insert(-1, pole -1j*r)
        line_segment_points.insert(-1, pole +1j*r)
    
    # alles Zusammenfgen
    all_segments = []
    for i in range(int(len(line_segment_points)/2)):
        p1 = line_segment_points[2*i]
        p2 = line_segment_points[2*i+1]
        segment = np.linspace(p1, p2, n)
        
        # Geradensegment anfgen
        all_segments.append(segment)
        
        # Halbkreissegment anfgen
        all_segments.append(semi_circles[i])
        
    return all_segments

# Plotten vorbereiten
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 8));

# Segmente erstellen
path_segments = create_nyquist_path_segments(P, r=0.5, R=10)
# path_segments -> Liste von Arrays mit komplexen Zahlen

for pseg in path_segments:
    # pseg -> Array mit komplexen Zahlen
    
    # Nyquist-Pfad plotten
    ax1.plot(np.real(pseg), np.imag(pseg))
    
    # Bild des Nyquistpfades unter Abbildung P(s) berechnen
    image_curve = control.evalfr(P, pseg) # Funktion fr 
    
    # ... und ploten
    ax2.plot(np.real(image_curve), np.imag(image_curve))
    
# Umlaufrichtung einzeichnen
util.drawCirc(ax1, radius=4, centX=3, centY=-5, startangle=0, diffangle=-300)
util.drawCirc(ax2, radius=0.15, centX=-0.18, centY=-0.2, startangle=0, diffangle=300)

ax1.set_title("Nyquistpfad")
ax2.set_title("Bild des Nyquistpfades unter $P(s)$")
ax1.axis("equal")
ax2.axis("equal")
plt.savefig('Nyquistpfad_und_Bild.pdf')
plt.show()

Segmentreihenfolge des Nyquistpfades: blau, orange, grn, rot.

Gut erkennbar ist, dass der kleine Halbkreis nahe des Pols bei $0 + 0j$ (orange, links) groe Funktionswerte erzeugt (orange, rechts), whrend der groe Halbkreis (rot, links), der weit entfernt von allen Polstellen ist, nur kleine Funktionswerte erzeugt (rot, rechts).
